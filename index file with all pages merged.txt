  <script>
    AOS.init({ once: true, duration: 700, easing: 'ease-out' });

    function toggleText(id) {
      const t = document.getElementById(id);
      document.querySelectorAll('.icon-text')
              .forEach(el => el !== t && (el.style.display = 'none'));
      t.style.display = t.style.display === 'block' ? 'none' : 'block';
    }

    const sections = document.querySelectorAll('section[id]');
    const links    = document.querySelectorAll('.nav-bar a[href^="#"]');
    window.addEventListener('scroll', () => {
      let y = window.pageYOffset;
      sections.forEach(s => {
        if (y >= s.offsetTop - 80 && y < s.offsetTop + s.offsetHeight) {
          links.forEach(a => a.classList.remove('active'));
          document.querySelector(`.nav-bar a[href="#${s.id}"]`)
                  .classList.add('active');
        }
      });
    });
